Introduction to Java Swing

Swing is Java's primary GUI (Graphical User Interface) toolkit, providing a rich set of widgets and components for building desktop applications with sophisticated user interfaces.

Swing vs AWT:

/*
 * SWING vs AWT COMPARISON
 * 
 * ┌────────────────────────────────────────────────────────────┐
 * │                    GUI TOOLKITS IN JAVA                    │
 * ├────────────────────────────────────────────────────────────┤
 * │                                                            │
 * │  AWT (Abstract Window Toolkit)                             │
 * │  - Original Java GUI toolkit                               │
 * │  - Uses native OS components (heavyweight)                 │
 * │  - Platform-dependent look and feel                        │
 * │  - Limited components                                      │
 * │                                                            │
 * │  SWING                                                     │
 * │  - Built on top of AWT                                     │
 * │  - Pure Java components (lightweight)                      │
 * │  - Consistent look across platforms                        │
 * │  - Pluggable Look and Feel                                 │
 * │  - More components and features                            │
 * │  - All classes start with "J" (JFrame, JButton, etc.)     │
 * │                                                            │
 * └────────────────────────────────────────────────────────────┘
 * 
 * SWING COMPONENT HIERARCHY:
 * 
 *   java.lang.Object
 *       └── java.awt.Component
 *             └── java.awt.Container
 *                   └── javax.swing.JComponent
 *                         ├── JLabel
 *                         ├── JButton
 *                         ├── JTextField
 *                         ├── JTextArea
 *                         ├── JPanel
 *                         ├── JComboBox
 *                         └── ... (many more)
 */

Creating Your First Swing Window:

// =============================================
// SWING BASICS - FIRST WINDOW
// =============================================

import javax.swing.*;
import java.awt.*;

public class FirstSwingWindow {
    
    public static void main(String[] args) {
        // ===== METHOD 1: Simple Direct Creation =====
        
        // Create a JFrame - the main window container
        JFrame frame = new JFrame("My First Swing Window");
        
        // Set the size of the window (width x height in pixels)
        frame.setSize(500, 400);
        
        // What happens when user clicks the X button?
        // EXIT_ON_CLOSE: Terminates the application
        // HIDE_ON_CLOSE: Hides the window but app continues
        // DISPOSE_ON_CLOSE: Destroys only this window
        // DO_NOTHING_ON_CLOSE: Does nothing (handle manually)
        frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        
        // Center the window on screen
        frame.setLocationRelativeTo(null);
        
        // Create a label and add it to the frame
        JLabel label = new JLabel("Welcome to Swing!", JLabel.CENTER);
        label.setFont(new Font("Arial", Font.BOLD, 24));
        frame.add(label);
        
        // Make the frame visible (always do this last!)
        frame.setVisible(true);
    }
}

// =============================================
// BETTER APPROACH: Extending JFrame
// =============================================

class MyWindow extends JFrame {
    
    public MyWindow() {
        // Call parent constructor with window title
        super("My Custom Window");
        
        // Initialize the window
        initializeWindow();
        
        // Add components
        addComponents();
    }
    
    private void initializeWindow() {
        setSize(600, 450);
        setDefaultCloseOperation(EXIT_ON_CLOSE);
        setLocationRelativeTo(null);
        setResizable(true);  // Allow resizing
        
        // Set minimum size
        setMinimumSize(new Dimension(400, 300));
        
        // Set application icon (if you have an image)
        // setIconImage(new ImageIcon("icon.png").getImage());
    }
    
    private void addComponents() {
        // Use a panel for better organization
        JPanel mainPanel = new JPanel();
        mainPanel.setLayout(new BorderLayout());
        mainPanel.setBackground(new Color(240, 240, 245));
        
        // Title label
        JLabel titleLabel = new JLabel("Welcome to My Application");
        titleLabel.setFont(new Font("Segoe UI", Font.BOLD, 28));
        titleLabel.setHorizontalAlignment(JLabel.CENTER);
        titleLabel.setBorder(BorderFactory.createEmptyBorder(20, 0, 20, 0));
        
        // Subtitle
        JLabel subtitleLabel = new JLabel("Built with Java Swing");
        subtitleLabel.setFont(new Font("Segoe UI", Font.PLAIN, 14));
        subtitleLabel.setHorizontalAlignment(JLabel.CENTER);
        subtitleLabel.setForeground(Color.GRAY);
        
        // Add components to panel
        mainPanel.add(titleLabel, BorderLayout.CENTER);
        mainPanel.add(subtitleLabel, BorderLayout.SOUTH);
        
        // Add panel to frame
        add(mainPanel);
    }
    
    public static void main(String[] args) {
        // Best practice: Run GUI code on Event Dispatch Thread
        SwingUtilities.invokeLater(() -> {
            MyWindow window = new MyWindow();
            window.setVisible(true);
        });
    }
}

    private JPanel createDialogPanel() {
        JPanel panel = new JPanel(new FlowLayout(FlowLayout.CENTER, 20, 20));
        panel.setBorder(new EmptyBorder(20, 20, 20, 20));
        
        // Message dialogs
        JButton infoBtn = new JButton("Information Dialog");
        infoBtn.addActionListener(e -> {
            JOptionPane.showMessageDialog(this,
                "This is an information message.",
                "Information",
                JOptionPane.INFORMATION_MESSAGE);
        });
        
        JButton warnBtn = new JButton("Warning Dialog");
        warnBtn.addActionListener(e -> {
            JOptionPane.showMessageDialog(this,
                "This is a warning message!",
                "Warning",
                JOptionPane.WARNING_MESSAGE);
        });
        
        JButton errorBtn = new JButton("Error Dialog");
        errorBtn.addActionListener(e -> {
            JOptionPane.showMessageDialog(this,
                "An error has occurred!",
                "Error",
                JOptionPane.ERROR_MESSAGE);
        });
        
        JButton confirmBtn = new JButton("Confirm Dialog");
        confirmBtn.addActionListener(e -> {
            int result = JOptionPane.showConfirmDialog(this,
                "Are you sure you want to proceed?",
                "Confirm",
                JOptionPane.YES_NO_CANCEL_OPTION);
            
            String message = switch(result) {
                case JOptionPane.YES_OPTION -> "You clicked YES";
                case JOptionPane.NO_OPTION -> "You clicked NO";
                case JOptionPane.CANCEL_OPTION -> "You clicked CANCEL";
                default -> "Dialog was closed";
            };
            JOptionPane.showMessageDialog(this, message);
        });
        
        JButton inputBtn = new JButton("Input Dialog");
        inputBtn.addActionListener(e -> {
            String name = JOptionPane.showInputDialog(this,
                "Enter your name:",
                "Input Required",
                JOptionPane.QUESTION_MESSAGE);
            
            if (name != null && !name.trim().isEmpty()) {
                JOptionPane.showMessageDialog(this, "Hello, " + name + "!");
            }
        });
        
        panel.add(infoBtn);
        panel.add(warnBtn);
        panel.add(errorBtn);
        panel.add(confirmBtn);
        panel.add(inputBtn);
        
        return panel;
    }
    
    // Helper method to create titled sections
    private JPanel createSection(String title) {
        JPanel section = new JPanel();
        section.setLayout(new BoxLayout(section, BoxLayout.Y_AXIS));
        section.setBorder(BorderFactory.createTitledBorder(
            BorderFactory.createLineBorder(new Color(200, 200, 200)),
            title,
            TitledBorder.LEFT,
            TitledBorder.TOP,
            new Font("Arial", Font.BOLD, 12)
        ));
        section.setAlignmentX(Component.LEFT_ALIGNMENT);
        return section;
    }

Layout Managers - Complete Guide

Layout managers control how components are arranged within a container. Choosing the right layout manager is crucial for creating well-organized user interfaces.

// =============================================
// LAYOUT MANAGERS - COMPREHENSIVE GUIDE
// =============================================

import javax.swing.*;
import javax.swing.border.*;
import java.awt.*;

public class LayoutManagersDemo extends JFrame {
    
    public LayoutManagersDemo() {
        super("Layout Managers Demo");
        setSize(900, 700);
        setDefaultCloseOperation(EXIT_ON_CLOSE);
        setLocationRelativeTo(null);
        
        JTabbedPane tabs = new JTabbedPane();
        
        tabs.addTab("FlowLayout", createFlowLayoutDemo());
        tabs.addTab("BorderLayout", createBorderLayoutDemo());
        tabs.addTab("GridLayout", createGridLayoutDemo());
        tabs.addTab("BoxLayout", createBoxLayoutDemo());
        tabs.addTab("GridBagLayout", createGridBagLayoutDemo());
        tabs.addTab("CardLayout", createCardLayoutDemo());
        
        add(tabs);
    }
    
    // =============================================
    // 1. FLOWLAYOUT - Default for JPanel
    // =============================================
    private JPanel createFlowLayoutDemo() {
        JPanel container = new JPanel(new BorderLayout());
        container.setBorder(new EmptyBorder(10, 10, 10, 10));
        
        // Description
        JTextArea desc = new JTextArea(
            "FlowLayout:\n" +
            "• Default layout for JPanel\n" +
            "• Arranges components in a row, left to right\n" +
            "• Wraps to next row when no more room\n" +
            "• Components maintain preferred size\n" +
            "• Can align LEFT, CENTER, or RIGHT"
        );
        desc.setEditable(false);
        desc.setBackground(new Color(255, 255, 230));
        container.add(desc, BorderLayout.NORTH);
        
        // Demo panels
        JPanel demoPanel = new JPanel(new GridLayout(3, 1, 5, 5));
        
        // Left aligned
        JPanel leftPanel = new JPanel(new FlowLayout(FlowLayout.LEFT, 10, 10));
        leftPanel.setBorder(BorderFactory.createTitledBorder("FlowLayout.LEFT"));
        for (int i = 1; i <= 5; i++) {
            leftPanel.add(new JButton("Button " + i));
        }
        
        // Center aligned (default)
        JPanel centerPanel = new JPanel(new FlowLayout(FlowLayout.CENTER, 10, 10));
        centerPanel.setBorder(BorderFactory.createTitledBorder("FlowLayout.CENTER"));
        for (int i = 1; i <= 5; i++) {
            centerPanel.add(new JButton("Button " + i));
        }
        
        // Right aligned
        JPanel rightPanel = new JPanel(new FlowLayout(FlowLayout.RIGHT, 10, 10));
        rightPanel.setBorder(BorderFactory.createTitledBorder("FlowLayout.RIGHT"));
        for (int i = 1; i <= 5; i++) {
            rightPanel.add(new JButton("Button " + i));
        }
        
        demoPanel.add(leftPanel);
        demoPanel.add(centerPanel);
        demoPanel.add(rightPanel);
        
        container.add(demoPanel, BorderLayout.CENTER);
        
        return container;
    }

    private JPanel createGridLayoutDemo() {

            "• Components fill their cell completely\n" +
            "• Great for calculators, keypads, grids"

        desc.setBackground(new Color(255, 255, 230));
        container.add(desc, BorderLayout.NORTH);
        
        JPanel demos = new JPanel(new GridLayout(1, 2, 10, 10));

    private void registerStudent() {
        // Validate fields
        StringBuilder errors = new StringBuilder();
        
        if (firstNameField.getText().trim().isEmpty()) {
            errors.append("• First name is required\n");
        }
        if (lastNameField.getText().trim().isEmpty()) {
            errors.append("• Last name is required\n");
        }
        if (!maleRadio.isSelected() && !femaleRadio.isSelected() && 
            !otherRadio.isSelected()) {
            errors.append("• Gender is required\n");
        }
        if (emailField.getText().trim().isEmpty() || 
            !emailField.getText().contains("@")) {
            errors.append("• Valid email is required\n");
        }
        if (courseCombo.getSelectedIndex() == 0) {
            errors.append("• Course selection is required\n");
        }
        if (yearCombo.getSelectedIndex() == 0) {
            errors.append("• Year level is required\n");
        }
        
        String password = new String(passwordField.getPassword());
        String confirmPassword = new String(confirmPasswordField.getPassword());
        
        if (password.isEmpty()) {
            errors.append("• Password is required\n");
        } else if (password.length() < 6) {
            errors.append("• Password must be at least 6 characters\n");
        } else if (!password.equals(confirmPassword)) {
            errors.append("• Passwords do not match\n");
        }
        
        if (!termsCheck.isSelected()) {
            errors.append("• You must accept the Terms and Conditions\n");
        }
        
        // Show errors or register
        if (errors.length() > 0) {
            JOptionPane.showMessageDialog(this, 
                "Please fix the following errors:\n\n" + errors.toString(),
                "Validation Error", 
                JOptionPane.ERROR_MESSAGE);
            statusLabel.setText("Registration failed - please check errors");
            statusLabel.setForeground(Color.RED);
        } else {
            // Create student record
            String gender = maleRadio.isSelected() ? "Male" : 
                           femaleRadio.isSelected() ? "Female" : "Other";
            
            String record = String.format(
                "Name: %s %s | Age: %d | Gender: %s | Email: %s | " +
                "Course: %s | Year: %s | Registered: %s",
                firstNameField.getText().trim(),
                lastNameField.getText().trim(),
                (Integer) ageSpinner.getValue(),
                gender,
                emailField.getText().trim(),
                courseCombo.getSelectedItem(),
                yearCombo.getSelectedItem(),
                LocalDate.now().format(DateTimeFormatter.ofPattern("MMM dd, yyyy"))
            );
            
            registeredStudents.add(record);
            
            JOptionPane.showMessageDialog(this,
                "Registration successful!\n\nWelcome, " + 
                firstNameField.getText() + "!",
                "Success",
                JOptionPane.INFORMATION_MESSAGE);
            
            statusLabel.setText("✓ Student registered successfully");
            statusLabel.setForeground(new Color(46, 139, 87));
            
            clearForm();
        }
    }
    
    private void clearForm() {

        lastNameField.setText("");

        addressArea.setText("");
        passwordField.setText("");

    }
    
    private void viewRegistered() {
        if (registeredStudents.isEmpty()) {
            JOptionPane.showMessageDialog(this,
                "No students registered yet.",
                "Registered Students",
                JOptionPane.INFORMATION_MESSAGE);
            return;
        }
        
        StringBuilder sb = new StringBuilder();
        sb.append("═══════════════════════════════════════\n");
        sb.append("          REGISTERED STUDENTS          \n");
        sb.append("═══════════════════════════════════════\n\n");
        
        for (int i = 0; i < registeredStudents.size(); i++) {
            sb.append((i + 1) + ". " + registeredStudents.get(i) + "\n\n");
        }
        
        sb.append("Total: " + registeredStudents.size() + " student(s)");
        
        JTextArea textArea = new JTextArea(sb.toString());
        textArea.setEditable(false);
        textArea.setFont(new Font("Consolas", Font.PLAIN, 11));
        
        JScrollPane scrollPane = new JScrollPane(textArea);
        scrollPane.setPreferredSize(new Dimension(500, 300));
        
        JOptionPane.showMessageDialog(this, scrollPane,
            "Registered Students",
            JOptionPane.PLAIN_MESSAGE);
    }
