// =============================================
// BOOK.JAVA - MODEL CLASS

package model;

import java.io.Serializable;

/**
 * Represents a book in the library system.
 * Demonstrates encapsulation and data validation.
 */
public class Book implements Serializable {
    private static final long serialVersionUID = 1L;
    
    // Private fields (encapsulation)
    private String isbn;
    private String title;
    private String author;
    private String category;
    private int totalCopies;
    private int availableCopies;
    
    // Constructor with validation
    public Book(String isbn, String title, String author, 
                String category, int totalCopies) {
        setIsbn(isbn);
        setTitle(title);
        setAuthor(author);
        setCategory(category);
        setTotalCopies(totalCopies);
        this.availableCopies = totalCopies;
    }
    
    // Getters
    public String getIsbn() { return isbn; }
    public String getTitle() { return title; }
    public String getAuthor() { return author; }
    public String getCategory() { return category; }
    public int getTotalCopies() { return totalCopies; }
    public int getAvailableCopies() { return availableCopies; }
    
    // Setters with validation
    public void setIsbn(String isbn) {
        if (isbn == null || isbn.trim().isEmpty()) {
            throw new IllegalArgumentException("ISBN cannot be empty");
        }
        this.isbn = isbn.trim();
    }
    
    public void setTitle(String title) {
        if (title == null || title.trim().isEmpty()) {
            throw new IllegalArgumentException("Title cannot be empty");
        }
        this.title = title.trim();
    }
    
    public void setAuthor(String author) {
        if (author == null || author.trim().isEmpty()) {
            throw new IllegalArgumentException("Author cannot be empty");
        }
        this.author = author.trim();
    }
    
    public void setCategory(String category) {
        this.category = category != null ? category.trim() : "Uncategorized";
    }
    
    public void setTotalCopies(int totalCopies) {
        if (totalCopies < 1) {
            throw new IllegalArgumentException("Must have at least 1 copy");
        }
        this.totalCopies = totalCopies;
    }
    
    // Business logic methods
    public boolean isAvailable() {
        return availableCopies > 0;
    }
    
    public void borrowCopy() {
        if (availableCopies <= 0) {
            throw new IllegalStateException("No copies available");
        }
        availableCopies--;
    }
    
    public void returnCopy() {
        if (availableCopies >= totalCopies) {
            throw new IllegalStateException("All copies already returned");
        }
        availableCopies++;
    }
    
    // Convert to CSV format for file storage
    public String toCSV() {
        return String.format("%s,%s,%s,%s,%d,%d",
            isbn, title, author, category, totalCopies, availableCopies);
    }
    
    // Create from CSV line
    public static Book fromCSV(String csvLine) {
        String[] parts = csvLine.split(",");
        if (parts.length < 6) {
            throw new IllegalArgumentException("Invalid CSV format");
        }
        Book book = new Book(parts[0], parts[1], parts[2], parts[3],
                            Integer.parseInt(parts[4]));
        book.availableCopies = Integer.parseInt(parts[5]);
        return book;
    }
    
    @Override
    public String toString() {
        return String.format("[%s] %s by %s (%d/%d available)",
            isbn, title, author, availableCopies, totalCopies);
    }
    
    @Override
    public boolean equals(Object obj) {
        if (this == obj) return true;
        if (obj == null || getClass() != obj.getClass()) return false;
        Book book = (Book) obj;
        return isbn.equals(book.isbn);
    }
    
    @Override
    public int hashCode() {
        return isbn.hashCode();
    }
}

package model;

import java.io.Serializable;
import java.time.LocalDate;
import java.time.format.DateTimeFormatter;

/**
 * Abstract base class for library members.
 * Demonstrates inheritance and polymorphism.
 */
public abstract class Member implements Serializable {
    private static final long serialVersionUID = 1L;
    
    protected String memberId;
    protected String name;
    protected String email;
    protected String phone;
    protected LocalDate memberSince;
    protected int booksBorrowed;
    
    public Member(String memberId, String name, String email, String phone) {
        this.memberId = memberId;
        this.name = name;
        this.email = email;
        this.phone = phone;
        this.memberSince = LocalDate.now();
        this.booksBorrowed = 0;
    }
    
    // Abstract method - subclasses must implement
    public abstract int getMaxBooks();
    public abstract int getLoanDays();
    public abstract double getFinePerDay();
    public abstract String getMemberType();
    
    // Common methods
    public boolean canBorrow() {
        return booksBorrowed < getMaxBooks();
    }
    
    public void borrowBook() {
        if (!canBorrow()) {
            throw new IllegalStateException(
                "Maximum books (" + getMaxBooks() + ") already borrowed");
        }
        booksBorrowed++;
    }
    
    public void returnBook() {
        if (booksBorrowed > 0) {
            booksBorrowed--;
        }
    }
    
    // Getters
    public String getMemberId() { return memberId; }
    public String getName() { return name; }
    public String getEmail() { return email; }
    public String getPhone() { return phone; }
    public LocalDate getMemberSince() { return memberSince; }
    public int getBooksBorrowed() { return booksBorrowed; }
    
    @Override
    public String toString() {
        return String.format("[%s] %s (%s) - %d books borrowed",
            memberId, name, getMemberType(), booksBorrowed);
    }
}

    public BorrowingLimitException(String memberName, int limit) {

    }

// =============================================
// LIBRARY SERVICE - BUSINESS LOGIC LAYER

package service;

import model.*;
import exception.*;
import java.io.*;
import java.util.*;
import java.time.LocalDate;
import java.time.temporal.ChronoUnit;

/**
 * Main service class handling library operations.
 * Implements singleton pattern for single point of access.
 */
public class LibraryService {
    private static LibraryService instance;
    
    private Map books;
    private Map members;
    private List loans;
    private List loanHistory;
    
    private static final String BOOKS_FILE = "data/books.csv";
    private static final String MEMBERS_FILE = "data/members.csv";
    private static final String LOANS_FILE = "data/loans.csv";
    
    // Singleton pattern
    private LibraryService() {
        books = new HashMap<>();
        members = new HashMap<>();
        loans = new ArrayList<>();
        loanHistory = new ArrayList<>();
        loadData();
    }
    
    public static LibraryService getInstance() {
        if (instance == null) {
            instance = new LibraryService();
        }
        return instance;
    }
    
    // =============================================
    // BOOK OPERATIONS
    // =============================================
    
    public void addBook(Book book) {
        if (books.containsKey(book.getIsbn())) {
            throw new IllegalArgumentException(
                "Book with ISBN " + book.getIsbn() + " already exists");
        }
        books.put(book.getIsbn(), book);
        saveBooks();
    }
    
    public Book findBook(String isbn) throws BookNotFoundException {
        Book book = books.get(isbn);
        if (book == null) {
            throw new BookNotFoundException(isbn);
        }
        return book;
    }
    
    public List searchBooks(String keyword) {
        List results = new ArrayList<>();
        String lowerKeyword = keyword.toLowerCase();
        
        for (Book book : books.values()) {
            if (book.getTitle().toLowerCase().contains(lowerKeyword) ||
                book.getAuthor().toLowerCase().contains(lowerKeyword) ||
                book.getIsbn().toLowerCase().contains(lowerKeyword)) {
                results.add(book);
            }
        }
        return results;
    }
    
    public List getAllBooks() {
        return new ArrayList<>(books.values());
    }
    
    public List getAvailableBooks() {
        List available = new ArrayList<>();
        for (Book book : books.values()) {
            if (book.isAvailable()) {
                available.add(book);
            }
        }
        return available;
    }
    
    // =============================================
    // MEMBER OPERATIONS
    // =============================================
    
    public void addMember(Member member) {
        if (members.containsKey(member.getMemberId())) {
            throw new IllegalArgumentException(
                "Member with ID " + member.getMemberId() + " already exists");
        }
        members.put(member.getMemberId(), member);
        saveMembers();
    }
    
    public Member findMember(String memberId) throws MemberNotFoundException {
        Member member = members.get(memberId);
        if (member == null) {
            throw new MemberNotFoundException(memberId);
        }
        return member;
    }
    
    public List getAllMembers() {
        return new ArrayList<>(members.values());
    }
    
    // =============================================
    // LOAN OPERATIONS
    // =============================================
    
    public Loan borrowBook(String memberId, String isbn) 
            throws MemberNotFoundException, BookNotFoundException,
                   BookNotAvailableException, BorrowingLimitException {
        
        Member member = findMember(memberId);
        Book book = findBook(isbn);
        
        // Check if book is available
        if (!book.isAvailable()) {
            throw new BookNotAvailableException(book.getTitle());
        }
        
        // Check if member can borrow more books
        if (!member.canBorrow()) {
            throw new BorrowingLimitException(
                member.getName(), member.getMaxBooks());
        }
        
        // Create loan
        Loan loan = new Loan(member, book, member.getLoanDays());
        
        // Update book and member
        book.borrowCopy();
        member.borrowBook();
        loans.add(loan);
        
        // Save changes
        saveBooks();
        saveLoans();
        
        return loan;
    }
    
    public double returnBook(String loanId) 
            throws OverdueLoanException {
        
        Loan loan = null;
        for (Loan l : loans) {
            if (l.getLoanId().equals(loanId)) {
                loan = l;
                break;
            }
        }
        
        if (loan == null) {
            throw new IllegalArgumentException("Loan not found: " + loanId);
        }
        
        // Calculate fine if overdue
        double fine = 0;
        int daysOverdue = loan.getDaysOverdue();
        if (daysOverdue > 0) {
            fine = daysOverdue * loan.getMember().getFinePerDay();
        }
        
        // Update book and member
        loan.getBook().returnCopy();
        loan.getMember().returnBook();
        loan.setReturnDate(LocalDate.now());
        loan.setFineAmount(fine);
        
        // Move to history
        loans.remove(loan);
        loanHistory.add(loan);
        
        // Save changes
        saveBooks();
        saveLoans();
        
        if (fine > 0) {
            throw new OverdueLoanException(daysOverdue, fine);
        }
        
        return fine;
    }
    
    public List getActiveLoans() {
        return new ArrayList<>(loans);
    }
    
    public List getOverdueLoans() {
        List overdue = new ArrayList<>();
        for (Loan loan : loans) {
            if (loan.isOverdue()) {
                overdue.add(loan);
            }
        }
        return overdue;
    }
    
    public List getMemberLoans(String memberId) {
        List memberLoans = new ArrayList<>();
        for (Loan loan : loans) {
            if (loan.getMember().getMemberId().equals(memberId)) {
                memberLoans.add(loan);
            }
        }
        return memberLoans;
    }
    
    // =============================================
    // FILE I/O OPERATIONS
    // =============================================
    
    private void loadData() {
        loadBooks();
        loadMembers();
        loadLoans();
    }
    
    private void loadBooks() {
        File file = new File(BOOKS_FILE);
        if (!file.exists()) return;
        
        try (BufferedReader reader = new BufferedReader(new FileReader(file))) {
            String line;
            reader.readLine(); // Skip header
            while ((line = reader.readLine()) != null) {
                try {
                    Book book = Book.fromCSV(line);
                    books.put(book.getIsbn(), book);
                } catch (Exception e) {
                    System.err.println("Error parsing book: " + line);
                }
            }
            System.out.println("Loaded " + books.size() + " books");
        } catch (IOException e) {
            System.err.println("Error loading books: " + e.getMessage());
        }
    }
    
    private void saveBooks() {
        try {
            new File("data").mkdirs(); // Create directory if not exists
            try (PrintWriter writer = new PrintWriter(BOOKS_FILE)) {
                writer.println("ISBN,Title,Author,Category,TotalCopies,Available");
                for (Book book : books.values()) {
                    writer.println(book.toCSV());
                }
            }
        } catch (IOException e) {
            System.err.println("Error saving books: " + e.getMessage());
        }
    }
    
    private void loadMembers() {
        // Similar implementation for members
    }
    
    private void saveMembers() {
        // Similar implementation for members
    }
    
    private void loadLoans() {
        // Similar implementation for loans
    }
    
    private void saveLoans() {
        // Similar implementation for loans
    }
    
    // =============================================
    // STATISTICS
    // =============================================
    
    public Map getStatistics() {
        Map stats = new HashMap<>();
        stats.put("totalBooks", books.size());
        stats.put("totalMembers", members.size());
        stats.put("activeLoans", loans.size());
        stats.put("overdueLoans", getOverdueLoans().size());
        
        int totalCopies = 0;
        int availableCopies = 0;
        for (Book book : books.values()) {
            totalCopies += book.getTotalCopies();
            availableCopies += book.getAvailableCopies();
        }
        stats.put("totalCopies", totalCopies);
        stats.put("availableCopies", availableCopies);
        
        return stats;
    }
}

    private void loadData() {
        refreshBookTable();
        refreshMemberTable();
        refreshLoanTable();
    }
    
    private void refreshBookTable() {
        bookTableModel.setRowCount(0);
        for (Book book : service.getAllBooks()) {
            String status = book.isAvailable() ? "âœ… Available" : "âŒ All Borrowed";
            bookTableModel.addRow(new Object[]{
                book.getIsbn(), book.getTitle(), book.getAuthor(),
                book.getCategory(), book.getTotalCopies(),
                book.getAvailableCopies(), status
            });
        }
        updateStatus("Loaded " + service.getAllBooks().size() + " books");
    }
    
    private void refreshMemberTable() {
        memberTableModel.setRowCount(0);
        for (Member member : service.getAllMembers()) {
            memberTableModel.addRow(new Object[]{
                member.getMemberId(), member.getName(),
                member.getMemberType(), member.getEmail(),
                member.getBooksBorrowed(), member.getMaxBooks()
            });
        }
    }
    
    private void refreshLoanTable() {
        loanTableModel.setRowCount(0);
        for (Loan loan : service.getActiveLoans()) {
            String status = loan.isOverdue() ? "âš ï¸ OVERDUE" : "âœ“ Active";
            loanTableModel.addRow(new Object[]{
                loan.getLoanId(), loan.getMember().getName(),
                loan.getBook().getTitle(), loan.getBorrowDate(),
                loan.getDueDate(), status
            });
        }
    }
    
    private void searchBooks() {
        String keyword = bookSearchField.getText().trim();
        if (keyword.isEmpty()) {
            refreshBookTable();
            return;
        }
        
        bookTableModel.setRowCount(0);
        List results = service.searchBooks(keyword);
        for (Book book : results) {
            String status = book.isAvailable() ? "âœ… Available" : "âŒ All Borrowed";
            bookTableModel.addRow(new Object[]{
                book.getIsbn(), book.getTitle(), book.getAuthor(),
                book.getCategory(), book.getTotalCopies(),
                book.getAvailableCopies(), status
            });
        }
        updateStatus("Found " + results.size() + " books matching '" + keyword + "'");
    }
