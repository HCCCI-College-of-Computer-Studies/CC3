Lesson V

Functions with Arrays

Learn to pass arrays to functions and create reusable code for array processing and manipulation.

Lesson Overview

Combining functions with arrays creates powerful, reusable code for data processing. You can write functions that operate on arrays of any size, making your code more flexible and maintainable.

Real-World Context: Array functions are used everywhere: sorting student grades, filtering search results, calculating statistics from datasets, processing pixel data in images, and analyzing financial transactions.

Learning Outcomes

  • Pass one-dimensional arrays as parameters to functions
  • Return arrays from functions
  • Pass two-dimensional arrays to functions
  • Process and manipulate array data within functions

1. Functions with 1D Arrays

When you pass an array to a function in Java, you're passing a reference to the array. This means changes made inside the function affect the original array.

Passing Arrays to Functions

Java 
public class ArrayFunctions {
    
    // Function to print all array elements
    public static void printArray(int[] arr) {
        for (int element : arr) {
            System.out.print(element + " ");
        }
        System.out.println();
    }
    
    // Function to calculate sum
    public static int calculateSum(int[] arr) {
        int sum = 0;
        for (int num : arr) {
            sum += num;
        }
        return sum;
    }
    
    // Function to find maximum value
    public static int findMax(int[] arr) {
        int max = arr[0];
        for (int i = 1; i < arr.length; i++) {
            if (arr[i] > max) {
                max = arr[i];
            }
        }
        return max;
    }
    
    // Function to calculate average
    public static double calculateAverage(int[] arr) {
        return (double) calculateSum(arr) / arr.length;
    }
    
    public static void main(String[] args) {
        int[] scores = {85, 92, 78, 95, 88};
        
        System.out.print("Scores: ");
        printArray(scores);
        
        System.out.println("Sum: " + calculateSum(scores));
        System.out.println("Max: " + findMax(scores));
        System.out.printf("Average: %.2f%n", calculateAverage(scores));
    }
}

Modifying Arrays in Functions

Java 
public class ModifyArrays {
    
    // Double each element in the array
    public static void doubleElements(int[] arr) {
        for (int i = 0; i < arr.length; i++) {
            arr[i] *= 2;  // Modifies original array
        }
    }
    
    // Reverse array in place
    public static void reverseArray(int[] arr) {
        int left = 0;
        int right = arr.length - 1;
        
        while (left < right) {
            // Swap elements
            int temp = arr[left];
            arr[left] = arr[right];
            arr[right] = temp;
            left++;
            right--;
        }
    }
    
    public static void main(String[] args) {
        int[] numbers = {1, 2, 3, 4, 5};
        
        doubleElements(numbers);
        System.out.println("After doubling: " + java.util.Arrays.toString(numbers));
        
        reverseArray(numbers);
        System.out.println("After reversing: " + java.util.Arrays.toString(numbers));
    }
}

Returning Arrays from Functions

Java 
public class ReturnArrays {
    
    // Create and return a new array with squared values
    public static int[] squareArray(int[] arr) {
        int[] result = new int[arr.length];
        for (int i = 0; i < arr.length; i++) {
            result[i] = arr[i] * arr[i];
        }
        return result;
    }
    
    // Generate an array of first n even numbers
    public static int[] generateEvenNumbers(int n) {
        int[] evens = new int[n];
        for (int i = 0; i < n; i++) {
            evens[i] = (i + 1) * 2;
        }
        return evens;
    }
    
    public static void main(String[] args) {
        int[] original = {1, 2, 3, 4, 5};
        int[] squared = squareArray(original);
        
        System.out.println("Original: " + java.util.Arrays.toString(original));
        System.out.println("Squared: " + java.util.Arrays.toString(squared));
        
        int[] evens = generateEvenNumbers(5);
        System.out.println("First 5 evens: " + java.util.Arrays.toString(evens));
    }
}

2. Functions with 2D Arrays

Two-dimensional arrays can also be passed to and returned from functions, enabling operations on tabular data like matrices or grade books.

Processing 2D Arrays

Java 
public class Matrix2D {
    
    // Print 2D array in table format
    public static void printMatrix(int[][] matrix) {
        for (int[] row : matrix) {
            for (int element : row) {
                System.out.printf("%4d", element);
            }
            System.out.println();
        }
    }
    
    // Calculate sum of all elements
    public static int sumMatrix(int[][] matrix) {
        int sum = 0;
        for (int[] row : matrix) {
            for (int element : row) {
                sum += element;
            }
        }
        return sum;
    }
    
    // Calculate row sums
    public static int[] rowSums(int[][] matrix) {
        int[] sums = new int[matrix.length];
        for (int i = 0; i < matrix.length; i++) {
            for (int j = 0; j < matrix[i].length; j++) {
                sums[i] += matrix[i][j];
            }
        }
        return sums;
    }
    
    // Add two matrices
    public static int[][] addMatrices(int[][] a, int[][] b) {
        int rows = a.length;
        int cols = a[0].length;
        int[][] result = new int[rows][cols];
        
        for (int i = 0; i < rows; i++) {
            for (int j = 0; j < cols; j++) {
                result[i][j] = a[i][j] + b[i][j];
            }
        }
        return result;
    }
    
    public static void main(String[] args) {
        int[][] grades = {
            {85, 90, 78},
            {92, 88, 95},
            {76, 84, 89System.out.println("Grade Matrix:");
        printMatrix(grades);
        
        System.out.println("\nTotal sum: " + sumMatrix(grades));
        
        int[] studentTotals = rowSums(grades);
        System.out.println("Student totals: " + java.util.Arrays.toString(studentTotals));
    }
}

Exercise: Count Occurrences

Write a function that counts how many times a given value appears in an array:



Solution 
public static int countOccurrences(int[] arr, int target) {
    int count = 0;
    for (int num : arr) {
        if (num == target) {
            count++;
        }
    }
    return count;
}

Summary

  • Arrays are passed by reference; changes in functions affect the original array.
  • Functions can return newly created arrays.
  • Use array.length to make functions work with arrays of any size.
  • 2D arrays can be processed using nested loops in functions.
  • Breaking array operations into functions improves code reusability.
